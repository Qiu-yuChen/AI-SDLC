class CSVEngine:
    """
    CSV 数据引擎
    封装所有 CSV 文件的读写操作，提供统一的增删改查接口。
    使用线程锁保障并发安全，采用"先备份后写入"策略。
    """

    _locks: dict = {}
    _global_lock = threading.Lock()

    def __init__(self, data_dir: str):
        """
        初始化 CSV 引擎

        Args:
            data_dir: CSV 数据文件目录路径
        """
        self.data_dir = data_dir
        os.makedirs(data_dir, exist_ok=True)

    def _get_lock(self, table: str) -> threading.Lock:
        """获取指定表的线程锁"""
        with self._global_lock:
            if table not in self._locks:
                self._locks[table] = threading.Lock()
            return self._locks[table]

    def _get_file_path(self, table: str) -> str:
        """获取 CSV 文件路径"""
        return os.path.join(self.data_dir, f"{table}.csv")

    def _get_backup_path(self, table: str) -> str:
        """获取备份文件路径"""
        return os.path.join(self.data_dir, f"{table}.bak")

    def _ensure_file_exists(self, table: str, fields: list[str]) -> None:
        """确保 CSV 文件存在，如果不存在则创建并写入表头"""
        file_path = self._get_file_path(table)
        if not os.path.exists(file_path):
            with open(file_path, "w", newline="", encoding="utf-8") as f:
                writer = csv.writer(f)
                writer.writerow(fields)

    def _read_csv(self, table: str) -> tuple[list[str], list[list[str]]]:
        """
        读取 CSV 文件

        Returns:
            (字段列表, 数据行列表) 的元组
        """
        file_path = self._get_file_path(table)
        if not os.path.exists(file_path):
            return [], []

        with open(file_path, "r", encoding="utf-8") as f:
            reader = csv.reader(f)
            rows = list(reader)

        if not rows:
            return [], []

        fields = rows[0]
        data = rows[1:]
        return fields, data

    def _write_csv(self, table: str, fields: list[str], data: list[list[str]]) -> None:
        """写入 CSV 文件（先备份后写入）"""
        file_path = self._get_file_path(table)
        backup_path = self._get_backup_path(table)

        if os.path.exists(file_path):
            try:
                os.replace(file_path, backup_path)
            except OSError:
                pass

        try:
            with open(file_path, "w", newline="", encoding="utf-8") as f:
                writer = csv.writer(f)
                writer.writerow(fields)
                writer.writerows(data)
        except Exception:
            if os.path.exists(backup_path):
                os.replace(backup_path, file_path)
            raise

        if os.path.exists(backup_path):
            try:
                os.remove(backup_path)
            except OSError:
                pass

    def _rows_to_dicts(self, fields: list[str], rows: list[list[str]]) -> list[dict]:
        """将 CSV 行数据转换为字典列表"""
        result = []
        for row in rows:
            if len(row) == len(fields):
                record = {}
                for i, field in enumerate(fields):
                    record[field] = row[i]
                result.append(record)
        return result

    def read_all(self, table: str) -> list[dict]:
        """读取指定表的所有记录"""
        lock = self._get_lock(table)
        with lock:
            fields, data = self._read_csv(table)
            return self._rows_to_dicts(fields, data)

    def read_by_id(self, table: str, record_id: int) -> Optional[dict]:
        """根据 ID 读取单条记录"""
        records = self.read_all(table)
        for record in records:
            if int(record.get("id", 0)) == record_id:
                return record
        return None

    def query(self, table: str, filters: dict) -> list[dict]:
        """条件查询记录"""
        records = self.read_all(table)
        result = []
        for record in records:
            match = True
            for key, value in filters.items():
                if value is not None:
                    if key not in record:
                        match = False
                        break
                    record_value = record.get(key, "")
                    if isinstance(value, int):
                        try:
                            if int(record_value) != value:
                                match = False
                                break
                        except (ValueError, TypeError):
                            match = False
                            break
                    else:
                        if record_value != str(value):
                            match = False
                            break
            if match:
                result.append(record)
        return result

    def get_next_id(self, table: str) -> int:
        """获取下一个可用 ID（自增）"""
        records = self.read_all(table)
        if not records:
            return 1
        max_id = 0
        for record in records:
            try:
                rid = int(record.get("id", 0))
                if rid > max_id:
                    max_id = rid
            except (ValueError, TypeError):
                continue
        return max_id + 1

    def insert(self, table: str, record: dict) -> int:
        """
        插入新记录

        Returns:
            新记录的 ID
        """
        lock = self._get_lock(table)
        with lock:
            fields, data = self._read_csv(table)

            if not fields:
                fields = list(record.keys())
                if "id" in fields:
                    fields.remove("id")
                fields = ["id"] + fields

            new_id = record.get("id")
            if new_id is None:
                existing = self._rows_to_dicts(fields, data)
                max_id = 0
                for r in existing:
                    try:
                        rid = int(r.get("id", 0))
                        if rid > max_id:
                            max_id = rid
                    except (ValueError, TypeError):
                        continue
                new_id = max_id + 1
                record["id"] = new_id

            new_row = [str(record.get(field, "")) for field in fields]
            data.append(new_row)

            self._write_csv(table, fields, data)
            return int(new_id)

    def update(self, table: str, record_id: int, data_dict: dict) -> bool:
        """更新记录"""
        lock = self._get_lock(table)
        with lock:
            fields, data = self._read_csv(table)
            if not fields:
                return False

            updated = False
            for i, row in enumerate(data):
                if len(row) == len(fields) and int(row[0]) == record_id:
                    for key, value in data_dict.items():
                        if key in fields:
                            idx = fields.index(key)
                            row[idx] = str(value)
                    data[i] = row
                    updated = True
                    break

            if updated:
                self._write_csv(table, fields, data)
            return updated

    def delete(self, table: str, record_id: int) -> bool:
        """删除记录"""
        lock = self._get_lock(table)
        with lock:
            fields, data = self._read_csv(table)
            if not fields:
                return False

            new_data = []
            deleted = False
            for row in data:
                if len(row) == len(fields) and int(row[0]) == record_id:
                    deleted = True
                else:
                    new_data.append(row)

            if deleted:
                self._write_csv(table, fields, new_data)
            return deleted

    def count(self, table: str) -> int:
        """获取记录总数"""
        records = self.read_all(table)
        return len(records)

def now_str() -> str:
    """获取当前时间字符串，格式: YYYY-MM-DD HH:mm:ss"""
    return datetime.now().strftime("%Y-%m-%d %H:%M:%S")


def parse_datetime(dt_str: str) -> Optional[datetime]:
    """解析时间字符串"""
    if not dt_str:
        return None
    for fmt in ("%Y-%m-%d %H:%M:%S", "%Y-%m-%d %H:%M"):
        try:
            return datetime.strptime(dt_str, fmt)
        except ValueError:
            continue
    return None

class AuthService:
    """
    认证服务
    管理用户登录、登出、会话 Token。
    当前版本模拟 AD/LDAP 认证。
    """

    # 内存会话存储 {token: user_dict}
    _sessions: dict = {}

    def __init__(self, engine: Optional[CSVEngine] = None):
        """
        初始化认证服务

        Args:

        """
        self.engine = engine or get_csv_engine()

    def login(self, username: str, password: str) -> tuple[Optional[str], Optional[str]]:
        """
        用户登录
        模拟 AD/LDAP 认证：本地数据库校验用户存在即通过。

        Args:
            username: 用户名/AD账号
            password: 密码

        Returns:
            (token, error_message) 元组
            成功时 token 为字符串，error 为 None
            失败时 token 为 None，error 为错误消息
        """
        try:
            # 根据 ad_account 查找用户
            users = self.engine.query("users", {"ad_account": username})
            if not users:
                # 也支持按 name 查找
                users = self.engine.query("users", {"name": username})

            if not users:
                return None, "用户名或密码错误"

            user = users[0]

            # 模拟 AD 认证：这里简化处理，任何非空密码都通过
            # 实际项目中应调用 AD/LDAP 服务器验证
            if not password:
                return None, "密码不能为空"

            # 生成会话 Token
            token = str(uuid.uuid4())
            self._sessions[token] = user

            # 记录审计日志
            self.log_service.log_operation(
                user_id=int(user["id"]),
                op_type="login",
                details=json.dumps({"username": username}, ensure_ascii=False)
            )

            return token, None

        except Exception as e:
            return None, f"登录失败: {str(e)}"

    def logout(self, token: str) -> Optional[str]:
        """
        用户登出

        Args:
            token: 会话 Token

        Returns:
            错误消息，成功为 None
        """
        user = self._sessions.pop(token, None)
        if user:
            self.log_service.log_operation(
                user_id=int(user["id"]),
                op_type="logout",
                details="{}"
            )
        return None

    def get_current_user(self, token: str) -> Optional[dict]:
        """
        根据 Token 获取当前用户信息

        Args:
            token: 会话 Token

        Returns:
            用户信息字典，无效 Token 返回 None
        """
        return self._sessions.get(token)

    def require_auth(self, token: str) -> tuple[Optional[dict], Optional[str]]:
        """
        认证中间件：校验 Token 并返回用户

        Args:
            token: 会话 Token

        Returns:
            (user, error) 元组
        """
        if not token:
            return None, "未提供认证Token"
        user = self.get_current_user(token)
        if not user:
            return None, "Token无效或已过期，请重新登录"
        return user, None

    def require_admin(self, token: str) -> tuple[Optional[dict], Optional[str]]:
        """
        管理员权限校验

        Args:
            token: 会话 Token

        Returns:
            (user, error) 元组
        """
        user, error = self.require_auth(token)
        if error:
            return None, error
        if user.get("role") != "admin":
            return None, "权限不足，需要管理员权限"
        return user, None

    def require_approver(self, token: str) -> tuple[Optional[dict], Optional[str]]:
        """
        审批人权限校验

        Args:
            token: 会话 Token

        Returns:
            (user, error) 元组
        """
        user, error = self.require_auth(token)
        if error:
            return None, error
        # 审批人可以是 admin 或任何用户
        return user, None

        """根据 ID 获取用户"""
        return self.engine.read_by_id("users", user_id)

    def create_user(self, user_data: UserCreate, operator_id: int) -> tuple[Optional[dict], Optional[str]]:
        """
        创建用户

        Args:
            user_data: 用户创建请求数据
            operator_id: 操作人ID

        Returns:
            (created_user, error) 元组
        """
        try:
            # 检查 AD 账号是否已存在
            existing = self.engine.query("users", {"ad_account": user_data.ad_account})
            if existing:
                return None, "AD账号已存在"

            record = {
                "name": user_data.name,
                "department": user_data.department,
                "phone": user_data.phone or "",
                "role": user_data.role,
                "ad_account": user_data.ad_account,
                "approver_id": str(user_data.approver_id) if user_data.approver_id else "",
            }

            new_id = self.engine.insert("users", record)

            self.log_service.log_operation(
                user_id=operator_id,
                op_type="create_user",
                details=json.dumps({"user_id": new_id, "name": user_data.name}, ensure_ascii=False)
            )

            return self.engine.read_by_id("users", new_id), None

        except Exception as e:
            return None, f"创建用户失败: {str(e)}"

    def update_user(self, user_id: int, user_data: UserUpdate, operator_id: int) -> tuple[Optional[dict], Optional[str]]:
        """
        更新用户信息

        Args:
            user_id: 用户ID
            user_data: 更新数据
            operator_id: 操作人ID

        Returns:
            (updated_user, error) 元组
        """
        try:
            existing = self.engine.read_by_id("users", user_id)
            if not existing:
                return None, "用户不存在"

            update_dict = {}
            if user_data.name is not None:
                update_dict["name"] = user_data.name
            if user_data.department is not None:
                update_dict["department"] = user_data.department
            if user_data.phone is not None:
                update_dict["phone"] = user_data.phone
            if user_data.role is not None:
                update_dict["role"] = user_data.role
            if user_data.ad_account is not None:
                update_dict["ad_account"] = user_data.ad_account
            if user_data.approver_id is not None:
                update_dict["approver_id"] = str(user_data.approver_id)

            if not update_dict:
                return existing, None

            success = self.engine.update("users", user_id, update_dict)
            if not success:
                return None, "更新用户失败"

            self.log_service.log_operation(
                user_id=operator_id,
                op_type="update_user",
                details=json.dumps({"user_id": user_id}, ensure_ascii=False)
            )

            return self.engine.read_by_id("users", user_id), None

        except Exception as e:
            return None, f"更新用户失败: {str(e)}"

        """
        删除用户

            user_id: 用户ID
            operator_id: 操作人ID

        Returns:
            错误消息，成功为 None
        """

                details=json.dumps({"user_id": user_id}, ensure_ascii=False)
            )

            return None

        except Exception as e:
            return f"删除用户失败: {str(e)}"
