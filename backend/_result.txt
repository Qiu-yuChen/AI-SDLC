```python
"""
工具函数模块
提供时间处理、ID生成、数据校验等通用功能
"""

import re
from datetime import datetime, timedelta
from typing import Tuple, Optional


def generate_reservation_id(date_str: str, sequence: int) -> str:
    """
    生成预约ID
    
    Args:
        date_str: 日期字符串，格式 YYYY-MM-DD
        sequence: 序号
        
    Returns:
        预约ID，格式 RYYYYMMDD001
    """
    date_part = date_str.replace("-", "")
    return f"R{date_part}{sequence:03d}"


def generate_payment_id(date_str: str, sequence: int) -> str:
    """
    生成缴费ID
    
    Args:
        date_str: 日期字符串，格式 YYYY-MM-DD
        sequence: 序号
        
    Returns:
        缴费ID，格式 PYYYYMMDD001
    """
    date_part = date_str.replace("-", "")
    return f"P{date_part}{sequence:03d}"


def get_current_datetime() -> str:
    """
    获取当前时间字符串
    
    Returns:
        当前时间字符串，格式 YYYY-MM-DD HH:MM:SS
    """
    return datetime.now().strftime("%Y-%m-%d %H:%M:%S")


def get_current_date() -> str:
    """
    获取当前日期字符串
    
    Returns:
        当前日期字符串，格式 YYYY-MM-DD
    """
    return datetime.now().strftime("%Y-%m-%d")


def validate_license_plate(license_plate: str) -> Tuple[bool, str]:
    """
    校验车牌号格式
    
    Args:
        license_plate: 车牌号
        
    Returns:
        (是否有效, 错误信息)
    """
    # 普通车牌格式：省份简称+字母+5位数字或字母
    # 新能源车牌格式：省份简称+字母+6位数字或字母
    pattern = r'^[京津沪渝冀豫云辽黑湘皖鲁新苏浙赣鄂桂甘晋蒙陕吉闽贵粤青藏川宁琼使领][A-Z][A-Z0-9]{5,6}$'
    
    if not license_plate:
        return False, "车牌号不能为空"
    
    if re.match(pattern, license_plate):
        return True, ""
    
    return False, "车牌格式不正确，请按鲁G12345或鲁G12345E格式输入"


def validate_phone(phone: str) -> Tuple[bool, str]:
    """
    校验手机号格式
    
    Args:
        phone: 手机号
        
    Returns:
        (是否有效, 错误信息)
    """
    pattern = r'^1[3-9]\d{9}$'
    
    if not phone:
        return False, "手机号不能为空"
    
    if re.match(pattern, phone):
        return True, ""
    
    return False, "手机号格式不正确"


def validate_reservation_date(date_str: str) -> Tuple[bool, str]:
    """
    校验预约日期
    
    Args:
        date_str: 预约日期字符串，格式 YYYY-MM-DD
        
    Returns:
        (是否有效, 错误信息)
    """
    try:
        reservation_date = datetime.strptime(date_str, "%Y-%m-%d").date()
        today = datetime.now().date()
        max_date = today + timedelta(days=7)
        
        if reservation_date < today:
            return False, "预约日期不能早于今天"
        
        if reservation_date > max_date:
            return False, "只能预约未来7天内"
        
        return True, ""
    except ValueError:
        return False, "日期格式不正确，请使用 YYYY-MM-DD 格式"


def is_workday(date_str: str) -> bool:
    """
    判断是否为工作日（简单判断，不考虑节假日调休）
    
    Args:
        date_str: 日期字符串，格式 YYYY-MM-DD
        
    Returns:
        是否为工作日
    """
    try:
        date = datetime.strptime(date_str, "%Y-%m-%d")
        # 周一到周五为工作日（0-4），周六周日为休息日（5-6）
        return date.weekday() < 5
    except ValueError:
        return True


def validate_employee_id(employee_id: str) -> Tuple[bool, str]:
    """
    校验员工工号格式
    
    Args:
        employee_id: 员工工号
        
    Returns:
        (是否有效, 错误信息)
    """
    if not employee_id:
        return False, "员工工号不能为空"
    
    if len(employee_id) < 3 or len(employee_id) > 20:
        return False, "员工工号长度应在3-20个字符之间"
    
    return True, ""


def validate_parking_area(parking_area: str, valid_areas: list) -> Tuple[bool, str]:
    """
    校验园区名称
    
    Args:
        parking_area: 园区名称
        valid_areas: 有效园区列表
        
    Returns:
        (是否有效, 错误信息)
    """
    if not parking_area:
        return False, "园区名称不能为空"
    
    if parking_area not in valid_areas:
        return False, f"无效的园区名称，有效园区：{', '.join(valid_areas)}"
    
    return True, ""


def validate_amount(amount: int) -> Tuple[bool, str]:
    """
    校验缴费金额
    
    Args:
        amount: 缴费金额（单位：元）
        
    Returns:
        (是否有效, 错误信息)
    """
    if amount is None:
        return False, "缴费金额不能为空"
    
    if amount <= 0:
        return False, "缴费金额必须大于0"
    
    if amount > 10000:
        return False, "缴费金额超出限制"
    
    return True, ""


def format_response(code: int = 200, msg: str = "操作成功", data: Optional[dict] = None) -> dict:
    """
    格式化API响应
    
    Args:
        code: 状态码
        msg: 消息
        data: 数据
        
    Returns:
        标准响应字典
    """
    response = {
        "code": code,
        "msg": msg
    }
    if data is not None:
        response["data"] = data
    return response
```

```python
"""
数据模型模块
封装CSV文件的读写操作，支持增删改查
"""

import csv
import os
from typing import List, Optional, Dict, Any
from pathlib import Path
import threading

from utils import (
    generate_reservation_id,
    generate_payment_id,
    get_current_datetime,
    get_current_date,
    is_workday
)


# 数据目录路径
DATA_DIR = Path(__file__).parent / "data"


class CSVDatabase:
    """
    CSV数据库基类
    提供通用的CSV文件读写操作
    """
    
    def __init__(self, filename: str, fieldnames: List[str]):
        """
        初始化CSV数据库
        
        Args:
            filename: CSV文件名
            fieldnames: 字段名列表
        """
        self.filepath = DATA_DIR / filename
        self.fieldnames = fieldnames
        self._lock = threading.Lock()
        
        # 确保数据目录存在
        DATA_DIR.mkdir(parents=True, exist_ok=True)
        
        # 如果文件不存在，创建并写入表头
        if not self.filepath.exists():
            self._write_header()
    
    def _write_header(self) -> None:
        """写入CSV表头"""
        with open(self.filepath, 'w', newline='', encoding='utf-8') as f:
            writer = csv.DictWriter(f, fieldnames=self.fieldnames)
            writer.writeheader()
    
    def read_all(self) -> List[Dict[str, str]]:
        """
        读取所有记录
        
        Returns:
            记录列表
        """
        with self._lock:
            if not self.filepath.exists():
                return []
            
            with open(self.filepath, 'r', encoding='utf-8') as f:
                reader = csv.DictReader(f)
                return list(reader)
    
    def write_all(self, records: List[Dict[str, str]]) -> None:
        """
        写入所有记录（覆盖）
        
        Args:
            records: 记录列表
        """
        with self._lock:
            with open(self.filepath, 'w', newline='', encoding='utf-8') as f:
                writer = csv.DictWriter(f, fieldnames=self.fieldnames)
                writer.writeheader()
                writer.writerows(records)
    
    def append(self, record: Dict[str, str]) -> None:
        """
        追加一条记录
        
        Args:
            record: 记录字典
        """
        with self._lock:
            with open(self.filepath, 'a', newline='', encoding='utf-8') as f:
                writer = csv.DictWriter(f, fieldnames=self.fieldnames)
                writer.writerow(record)
    
    def find_by_field(self, field: str, value: str) -> List[Dict[str, str]]:
        """
        按字段查找记录
        
        Args:
            field: 字段名
            value: 字段值
            
        Returns:
            匹配的记录列表
        """
        records = self.read_all()
        return [r for r in records if r.get(field) == value]
    
    def find_one_by_field(self, field: str, value: str) -> Optional[Dict[str, str]]:
        """
        按字段查找单条记录
        
        Args:
            field: 字段名
            value: 字段值
            
        Returns:
            匹配的记录，如果不存在返回None
        """
        records = self.find_by_field(field, value)
        return records[0] if records else None
    
    def update_by_field(self, field: str, value: str, updates: Dict[str, str]) -> bool:
        """
        按字段更新记录
        
        Args:
            field: 字段名
            value: 字段值
            updates: 更新内容
            
        Returns:
            是否更新成功
        """
        records = self.read_all()
        updated = False
        
        for record in records:
            if record.get(field) == value:
                record.update(updates)
                updated = True
        
        if updated:
            self.write_all(records)
        
        return updated
    
    def delete_by_field(self, field: str, value: str) -> bool:
        """
        按字段删除记录
        
        Args:
            field: 字段名
            value: 字段值
            
        Returns:
            是否删除成功
        """
        records = self.read_all()
        new_records = [r for r in records if r.get(field) != value]
        
        if len(new_records) < len(records):
            self.write_all(new_records)
            return True
        
        return False

                'workday_quota': int(record['workday_quota']),
                'holiday_quota': int(record['holiday_quota']),
                'is_open': record['is_open'].lower() == 'true',
                'description': record.get('description', '')
            }
        return None

        更新园区配置
        
        Args:
            area: 园区名称

            holiday_quota: 节假日配额
            is_open: 是否开放
            
        Returns:
            是否更新成功
        """
        updates = {}
        if workday_quota is not None:
            updates['workday_quota'] = str(workday_quota)
        if holiday_quota is not None:
            updates['holiday_quota'] = str(holiday_quota)
        if is_open is not None:
            updates['is_open'] = 'true' if is_open else 'false'
        
        return self.update_by_field('area', area, updates)
    
    def get_quota(self, area: str, date_str: str) -> int:
        """
        获取指定日期的配额
        
        Args:
            area: 园区名称
            date_str: 日期字符串
            
        Returns:
            当日配额
        """
        config = self.get_config(area)
        if not config:
            return 0
        
        if is_workday(date_str):
            return config['workday_quota']
        else:
            return config['holiday_quota']
```

```python
class ReservationService:
    """
    预约服务类
    处理预约相关的业务逻辑
    """
    
    def __init__(self):
        """初始化预约服务"""
        self.park_model = park_model
        self.reservation_model = reservation_model
    
    def create_reservation(self, name: str, employee_id: str, phone: str,
                          parking_area: str, date: str, license_plate: str) -> Dict[str, Any]:
        """
        创建预约
        
        Args:
            name: 员工姓名
            employee_id: 员工工号
            phone: 手机号
            parking_area: 园区名称
            date: 预约日期
            license_plate: 车牌号
            
        Returns:
            响应结果
        """
        try:
            # 1. 校验员工姓名
            if not name or len(name.strip()) == 0:
                return format_response(400, "员工姓名不能为空")
            
            # 2. 校验员工工号
            valid, msg = validate_employee_id(employee_id)
            if not valid:
                return format_response(400, msg)
            
            # 3. 校验手机号
            valid, msg = validate_phone(phone)
            if not valid:
                return format_response(400, msg)
            
            # 4. 校验园区名称
            valid_areas = self.park_model.get_all_areas()
            valid, msg = validate_parking_area(parking_area, valid_areas)
            if not valid:
                return format_response(400, msg)
            
            # 5. 校验预约日期
            valid, msg = validate_reservation_date(date)
            if not valid:
                return format_response(400, msg)
            
            # 6. 校验车牌号
            valid, msg = validate_license_plate(license_plate)
            if not valid:
                return format_response(400, msg)
            
            # 7. 检查园区是否开放
            config = self.park_model.get_config(parking_area)
            if not config:
                return format_response(404, "园区不存在")
            
            if not config['is_open']:
                return format_response(400, "该园区暂未开放预约")
            
            # 8. 检查预约配额
            total_quota = self.park_model.get_quota(parking_area, date)
            current_count = self.reservation_model.get_remaining_quota(parking_area, date)
            
            if current_count >= total_quota:
                return format_response(400, "当前园区预约已满")
            
            # 9. 检查车牌是否已预约
            if self.reservation_model.check_license_exists(license_plate, date, parking_area):
                return format_response(400, "该车牌已在当前园区预约")
            
            # 10. 创建预约记录
            reservation = self.reservation_model.create_reservation(
                name=name.strip(),
                employee_id=employee_id.strip(),
                phone=phone.strip(),
                parking_area=parking_area,
                date=date,
                license_plate=license_plate.strip().upper()
            )
            
            # 11. 模拟推送到科拓系统
            self._push_to_ketuou_system(reservation)
            
            return format_response(
                200,
                "预约成功",
                {"reservation_id": reservation['reservation_id']}
            )
            
        except Exception as e:
            return format_response(500, f"系统错误：{str(e)}")
    
    def _push_to_ketuou_system(self, reservation: Dict[str, str]) -> bool:
        """
        模拟推送到科拓系统
        
        Args:
            reservation: 预约记录
            
        Returns:
            是否推送成功
        """
        # 这里模拟推送，实际项目中需要调用科拓系统API
        print(f"[科拓系统] 推送预约信息: {reservation}")
        return True
    
    def cancel_reservation(self, reservation_id: str) -> Dict[str, Any]:
        """
        取消预约
        
        Args:
            reservation_id: 预约ID
            
        Returns:
            响应结果
        """
        try:
            # 1. 检查预约是否存在
            reservation = self.reservation_model.get_reservation(reservation_id)
            if not reservation:
                return format_response(404, "预约记录不存在")
            
            # 2. 检查预约状态
            if reservation['status'] == '已取消':
                return format_response(400, "该预约已取消")
            
            # 3. 取消预约
            success = self.reservation_model.cancel_reservation(reservation_id)
            if not success:
                return format_response(500, "取消失败")
            
            # 4. 模拟通知科拓系统
            self._notify_cancel_to_ketuou(reservation)
            
            return format_response(200, "取消成功")
            
        except Exception as e:
            return format_response(500, f"系统错误：{str(e)}")
    
    def _notify_cancel_to_ketuou(self, reservation: Dict[str, str]) -> bool:
        """
        模拟通知科拓系统取消预约
        
        Args:
            reservation: 预约记录
            
        Returns:
            是否通知成功
        """
        print(f"[科拓系统] 通知取消预约: {reservation}")
        return True
    
    def get_reservation(self, reservation_id: str) -> Dict[str, Any]:
        """
        获取预约详情
        
        Args:
            reservation_id: 预约ID
            
        Returns:
            响应结果
        """
        try:
            reservation = self.reservation_model.get_reservation(reservation_id)
            if not reservation:
                return format_response(404, "预约记录不存在")
            
            return format_response(200, "查询成功", reservation)
            
        except Exception as e:
            return format_response(500, f"系统错误：{str(e)}")
    
    def get_remaining_quota(self, parking_area: str, date: str) -> Dict[str, Any]:
        """
        获取剩余车位数量
        
        Args:
            parking_area: 园区名称
            date: 日期
            
        Returns:
            响应结果
        """
        try:
            # 获取园区配置
            config = self.park_model.get_config(parking_area)
            if not config:
                return format_response(404, "园区不存在")
            
            # 获取总配额
            total_quota = self.park_model.get_quota(parking_area, date)
            
            # 获取已预约数量
            used_count = self.reservation_model.get_remaining_quota(parking_area, date)
            
            # 计算剩余数量
            remaining = total_quota - used_count
            
            return format_response(200, "查询成功", {
                "total_quota": total_quota,
                "used_count": used_count,
                "remaining": max(0, remaining),
                "is_open": config['is_open']
            })
            
        except Exception as e:
            return format_response(500, f"系统错误：{str(e)}")
```

```python
from pathlib import Path
from contextlib import asynccontextmanager

from fastapi import FastAPI, Request
from fastapi.staticfiles import StaticFiles
from fastapi.responses import FileResponse
from fastapi.middleware.cors import CORSMiddleware

from routes import router
from utils import format_response


# 项目根目录
BASE_DIR = Path(__file__).parent


@asynccontextmanager
async def lifespan(app: FastAPI):
    """
    应用生命周期管理
    启动时初始化数据目录，关闭时清理资源
    """
    # 启动时初始化
    print("🚀 员工临时车辆预约系统启动中...")
    
    # 确保数据目录存在
    data_dir = BASE_DIR / "data"
    data_dir.mkdir(parents=True, exist_ok=True)
    print(f"📁 数据目录: {data_dir}")
    
    # 确保静态文件目录存在
    static_dir = BASE_DIR / "static"
    static_dir.mkdir(parents=True, exist_ok=True)
    print(f"📁 静态文件目录: {static_dir}")
    
    print("✅ 系统初始化完成")
    
    yield
    
    # 关闭时清理
    print("👋 员工临时车辆预约系统已关闭")


# 创建FastAPI应用
app = FastAPI(
    title="员工临时车辆预约系统",
    description="""
## 员工临时车辆预约系统 API

提供以下功能：
- 临时车辆预约申请、取消
- 提前缴费功能
- 园区配置管理
- 剩余车位查询

### 园区列表
- 青岛
- 潍坊
- 荣成
- 东莞
    """,
    version="1.0.0",
    lifespan=lifespan,
    docs_url="/docs",
    redoc_url="/redoc"
)


# 配置CORS中间件
app.add_middleware(
    CORSMiddleware,
    allow_origins=["*"],  # 生产环境应限制具体域名
    allow_credentials=True,
    allow_methods=["*"],
    allow_headers=["*"],
)


# 注册API路由
app.include_router(router, prefix="/api")


# 挂载静态文件目录
static_dir = BASE_DIR / "static"
if static_dir.exists():
    app.mount("/static", StaticFiles(directory=str(static_dir)), name="static")


# 首页路由
@app.get("/", summary="首页", tags=["页面"])
async def index():
    """
    返回前端首页
    """
    index_file = static_dir / "index.html"
    if index_file.exists():
        return FileResponse(str(index_file))
    return {"message": "员工临时车辆预约系统 API", "docs": "/docs"}


# 全局异常处理
@app.exception_handler(Exception)
async def global_exception_handler(request: Request, exc: Exception):
    """
    全局异常处理器
    捕获所有未处理的异常，返回友好的错误信息
    """
    print(f"❌ 未处理的异常: {type(exc).__name__}: {exc}")
    return format_response(500, f"服务器内部错误，请稍后重试")


# 404处理
@app.exception_handler(404)
async def not_found_handler(request: Request, exc):
    """
    404错误处理器
    """
    return format_response(404, "请求的资源不存在")


if __name__ == "__main__":
    import uvicorn
    
    # 启动服务器
    uvicorn.run(
        "app:app",
        host="0.0.0.0",
        port=8000,
        reload=True,
        log_level="info"
    )
```
