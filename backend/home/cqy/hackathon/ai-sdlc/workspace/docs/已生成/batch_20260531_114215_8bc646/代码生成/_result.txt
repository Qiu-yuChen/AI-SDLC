```python
"""
工具函数模块

提供 CSV 文件操作的基础封装，包括增删改查等通用功能。
所有数据操作均基于 Python csv 标准库实现。
"""

import csv
import os
from typing import Any, Dict, List, Optional, Sequence
from datetime import datetime


class CsvDatabase:
    """
    CSV 数据操作封装类。

    提供对 CSV 文件的增删改查操作，模拟简单数据库行为。
    每个 CSV 文件对应一张"表"，首行为字段名。
    """

    def __init__(self, data_dir: str) -> None:
        """
        初始化 CsvDatabase 实例。

        Args:
            data_dir: CSV 数据文件存放目录路径
        """
        self.data_dir = data_dir
        os.makedirs(self.data_dir, exist_ok=True)

    def _get_file_path(self, table_name: str) -> str:
        """
        获取指定表对应的 CSV 文件完整路径。

        Args:
            table_name: 表名（不含扩展名）

        Returns:
            CSV 文件的完整路径
        """
        return os.path.join(self.data_dir, f"{table_name}.csv")

    def _ensure_file_exists(self, table_name: str, fields: List[str]) -> None:
        """
        确保 CSV 文件存在，若不存在则创建并写入表头。

        Args:
            table_name: 表名
            fields: 表头字段列表
        """
        file_path = self._get_file_path(table_name)
        if not os.path.exists(file_path):
            with open(file_path, mode="w", newline="", encoding="utf-8") as f:
                writer = csv.writer(f)
                writer.writerow(fields)

    def get_next_id(self, table_name: str) -> int:
        """
        获取指定表的下一个自增 ID。

        Args:
            table_name: 表名

        Returns:
            下一个可用 ID（从 1 开始）
        """
        records = self.select_all(table_name)
        if not records:
            return 1
        ids = [int(r.get("id", 0)) for r in records if r.get("id", "").isdigit()]
        return max(ids) + 1 if ids else 1

    def select_all(self, table_name: str) -> List[Dict[str, str]]:
        """
        查询指定表的所有记录。

        Args:
            table_name: 表名

        Returns:
            记录列表，每条记录为一个字典
        """
        file_path = self._get_file_path(table_name)
        if not os.path.exists(file_path):
            return []
        try:
            with open(file_path, mode="r", newline="", encoding="utf-8") as f:
                reader = csv.DictReader(f)
                return [row for row in reader]
        except Exception as e:
            print(f"[CSV Error] 读取 {table_name} 失败: {e}")
            return []

    def select_by(self, table_name: str, **kwargs: Any) -> List[Dict[str, str]]:
        """
        按条件查询记录（等值匹配）。

        Args:
            table_name: 表名
            **kwargs: 查询条件，如 location='WEF', status='pending'

        Returns:
            符合条件的记录列表
        """
        records = self.select_all(table_name)

        def match(row: Dict[str, str]) -> bool:
            for key, value in kwargs.items():
                if key not in row:
                    return False
                if str(value) != row[key]:
                    return False
            return True

        return [r for r in records if match(r)]

    def insert(self, table_name: str, fields: List[str], record: Dict[str, Any]) -> bool:
        """
        插入一条新记录。

        Args:
            table_name: 表名
            fields: 表头字段列表
            record: 要插入的数据字典

        Returns:
            插入成功返回 True，失败返回 False
        """
        try:
            self._ensure_file_exists(table_name, fields)
            file_path = self._get_file_path(table_name)
            with open(file_path, mode="a", newline="", encoding="utf-8") as f:
                writer = csv.DictWriter(f, fieldnames=fields)
                writer.writerow(record)
            return True
        except Exception as e:
            print(f"[CSV Error] 插入 {table_name} 失败: {e}")
            return False

    def update_by(
        self,
        table_name: str,
        condition: Dict[str, str],
        updates: Dict[str, Any],
    ) -> int:
        """
        按条件更新记录。

        Args:
            table_name: 表名
            condition: 更新条件，如 {"id": "1"}
            updates: 要更新的字段键值对

        Returns:
            更新的记录数量
        """
        records = self.select_all(table_name)
        if not records:
            return 0

        updated_count = 0
        fields = list(records[0].keys()) if records else []

        for record in records:
            match = True
            for k, v in condition.items():
                if record.get(k) != str(v):
                    match = False
                    break
            if match:
                for k, v in updates.items():
                    record[k] = str(v)
                updated_count += 1

        if updated_count > 0:
            try:
                file_path = self._get_file_path(table_name)
                with open(file_path, mode="w", newline="", encoding="utf-8") as f:
                    writer = csv.DictWriter(f, fieldnames=fields)
                    writer.writeheader()
                    writer.writerows(records)
            except Exception as e:
                print(f"[CSV Error] 更新 {table_name} 失败: {e}")
                return 0

        return updated_count

    def delete_by(self, table_name: str, condition: Dict[str, str]) -> int:
        """
        按条件删除记录。

        Args:
            table_name: 表名
            condition: 删除条件，如 {"id": "1"}

        Returns:
            删除的记录数量
        """
        records = self.select_all(table_name)
        if not records:
            return 0

        original_count = len(records)
        fields = list(records[0].keys()) if records else []

        records = [
            r for r in records
            if not all(r.get(k) == str(v) for k, v in condition.items())
        ]
        deleted_count = original_count - len(records)

        if deleted_count > 0:
            try:
                file_path = self._get_file_path(table_name)
                with open(file_path, mode="w", newline="", encoding="utf-8") as f:
                    writer = csv.DictWriter(f, fieldnames=fields)
                    writer.writeheader()
                    writer.writerows(records)
            except Exception as e:
                print(f"[CSV Error] 删除 {table_name} 失败: {e}")
                return 0

        return deleted_count

    获取当前时间的格式化字符串。

    """
    获取当前日期的格式化字符串。

    """
    return datetime.now().strftime("%Y-%m-%d")


# 工具函数：校验车牌号格式
def validate_license_plate(plate: str) -> bool:
    """
    校验车牌号格式是否合法。

    支持燃油车牌（如：鲁G12345）和新能源车牌（如：鲁G12345E）。

    Args:
        plate: 车牌号字符串

    Returns:
        合法返回 True，否则返回 False
    """
    if not plate or len(plate) < 6:
        return False
    # 简单校验：以汉字开头，包含字母和数字
    # 中文车牌常见格式：省份简称(1位) + 字母(1位) + 数字/字母(5-6位)
    import re
    pattern = r"^[\u4e00-\u9fa5][A-Za-z0-9][A-Za-z0-9]{4,6}$"
    return bool(re.match(pattern, plate.strip()))


# 工具函数：校验手机号格式
def validate_phone(phone: str) -> bool:
    """
    校验手机号格式是否合法（中国大陆手机号）。

    Args:
        phone: 手机号字符串

    Returns:
        合法返回 True，否则返回 False
    """
    import re
    pattern = r"^1[3-9]\d{9}$"
    return bool(re.match(pattern, phone.strip()))
```

```python
数据模型模块

定义系统数据模型（园区配置、预约记录、缴费记录），

class LocationConfig:
    """
    园区配置数据模型。

    管理各园区的启用/禁用状态及每日预约配额。
    CSV 文件: config_locations.csv
    字段: id, name, code, status, weekday_quota, weekend_quota
    """

    FIELDS = ["id", "name", "code", "status", "weekday_quota", "weekend_quota"]

    def __init__(self, db: CsvDatabase) -> None:
        """
        初始化 LocationConfig。

        Args:
            db: CsvDatabase 实例
        """
        self.db = db
        self._ensure_default_data()

    def _ensure_default_data(self) -> None:
        """
        确保默认的园区配置数据存在。
        若 CSV 文件为空，则插入默认的四个园区配置。
        """
        records = self.db.select_all("config_locations")
        if not records:
            defaults = [
                {"id": "1", "name": "潍坊", "code": "WEF", "status": "active",
                 "weekday_quota": "20", "weekend_quota": "10"},
                {"id": "2", "name": "青岛", "code": "QDG", "status": "active",
                 "weekday_quota": "15", "weekend_quota": "8"},
                {"id": "3", "name": "荣成", "code": "RCE", "status": "active",
                 "weekday_quota": "10", "weekend_quota": "5"},
                {"id": "4", "name": "东莞", "code": "DGW", "status": "active",
                 "weekday_quota": "25", "weekend_quota": "12"},
            ]
            for rec in defaults:
                self.db.insert("config_locations", self.FIELDS, rec)

    def get_all_locations(self) -> List[Dict[str, str]]:
        """
        获取所有园区配置。

        Returns:
            园区配置列表
        """
        return self.db.select_all("config_locations")

    def get_active_locations(self) -> List[Dict[str, str]]:
        """
        获取所有已启用的园区配置。

        Returns:
            状态为 active 的园区配置列表
        """
        return self.db.select_by("config_locations", status="active")

    def get_location_by_code(self, code: str) -> Optional[Dict[str, str]]:
        """
        根据园区编码获取配置信息。

        Args:
            code: 园区编码（如 WEF）

        Returns:
            园区配置字典，不存在则返回 None
        """
        records = self.db.select_by("config_locations", code=code)
        return records[0] if records else None

    def get_quota(self, code: str, date_str: Optional[str] = None) -> int:
        """
        获取指定园区在指定日期的最大预约配额。

        Args:
            code: 园区编码
            date_str: 日期字符串（YYYY-MM-DD），默认为当天

        Returns:
            最大预约配额数
        """
        if date_str is None:
            date_str = get_today_str()

        location = self.get_location_by_code(code)
        if location is None:
            return 0

        # 判断是工作日还是休息日
        dt = datetime.strptime(date_str, "%Y-%m-%d")
        is_weekend = dt.weekday() >= 5  # 周六(5)或周日(6)

        quota_str = location.get("weekend_quota" if is_weekend else "weekday_quota", "0")
        try:
            return int(quota_str)
        except (ValueError, TypeError):
            return 0

    def update_quota(self, code: str, weekday_quota: int, weekend_quota: int) -> bool:
        """
        更新指定园区的配额设置。

        Args:
            code: 园区编码
            weekday_quota: 工作日配额
            weekend_quota: 休息日配额

        Returns:
            更新成功返回 True
        """
        result = self.db.update_by(
            "config_locations",
            {"code": code},
            {"weekday_quota": str(weekday_quota), "weekend_quota": str(weekend_quota)},
        )
        return result > 0

    def toggle_status(self, code: str) -> bool:
        """
        切换指定园区的启用/禁用状态。

        Args:
            code: 园区编码

        Returns:
            切换成功返回 True
        """
        location = self.get_location_by_code(code)
        if location is None:
            return False
        new_status = "inactive" if location["status"] == "active" else "active"
        result = self.db.update_by(
            "config_locations",
            {"code": code},
            {"status": new_status},
        )
        return result > 0

# 预约记录模型

class ReservationRecord:
    """
    预约记录数据模型。

    管理员工车辆预约记录的增删改查。
    CSV 文件: reservation_records.csv
    字段: id, license_plate, location, date, status, note, created_at
    """

    FIELDS = ["id", "license_plate", "location", "date", "status", "note", "created_at"]

    def __init__(self, db: CsvDatabase) -> None:
        """
        初始化 ReservationRecord。

        Args:
            db: CsvDatabase 实例
        """
        self.db = db

    def create(
        self,
        license_plate: str,
        location: str,
        date: str,
        name: str,
        employee_id: str,
        phone: str,
    ) -> Tuple[bool, str]:
        """
        创建一条新的预约记录。

        Args:
            license_plate: 车牌号
            location: 园区编码
            date: 预约日期（YYYY-MM-DD）
            name: 员工姓名
            employee_id: 工号
            phone: 手机号

        Returns:
            (是否成功, 预约ID或错误信息)
        """
        # 检查同一车牌同一天是否已有预约
        existing = self.db.select_by(
            "reservation_records",
            license_plate=license_plate,
            date=date,
        )
        # 排除已取消的记录
        active = [r for r in existing if r.get("status") in ("pending", "confirmed")]
        if active:
            return False, "该车辆当天已有预约记录，无法重复预约"

        record_id = str(self.db.get_next_id("reservation_records"))
        note = f"{name}/{employee_id}/{phone}"
        now = get_now_str()

        record = {
            "id": record_id,
            "license_plate": license_plate,
            "location": location,
            "date": date,
            "status": "pending",  # 待生效
            "note": note,
            "created_at": now,
        }

        success = self.db.insert("reservation_records", self.FIELDS, record)
        if success:
            return True, record_id
        return False, "写入数据失败，请稍后重试"

    def cancel(self, reservation_id: str, license_plate: str) -> bool:
        """
        取消预约（将状态更新为 'canceled'）。

        Args:
            reservation_id: 预约记录 ID
            license_plate: 车牌号（用于二次校验）

        Returns:
            取消成功返回 True
        """
        result = self.db.update_by(
            "reservation_records",
            {"id": reservation_id, "license_plate": license_plate},
            {"status": "canceled"},
        )
        return result > 0

    def get_by_id(self, reservation_id: str) -> Optional[Dict[str, str]]:
        """
        根据预约 ID 获取预约记录。

        Args:
            reservation_id: 预约记录 ID

        Returns:
            预约记录字典，不存在则返回 None
        """
        records = self.db.select_by("reservation_records", id=reservation_id)
        return records[0] if records else None

    def get_by_plate(self, license_plate: str) -> List[Dict[str, str]]:
        """
        根据车牌号获取所有预约记录。

        Args:
            license_plate: 车牌号

        Returns:
            预约记录列表
        """
        return self.db.select_by("reservation_records", license_plate=license_plate)

    def get_by_employee(self, employee_id: str) -> List[Dict[str, str]]:
        """
        根据工号获取所有预约记录（通过 note 字段模糊匹配）。

        Args:
            employee_id: 员工工号

        Returns:
            预约记录列表
        """
        all_records = self.db.select_all("reservation_records")
        return [r for r in all_records if f"/{employee_id}/" in r.get("note", "")]

    def get_by_location_date(
        self, location: str, date: str
    ) -> List[Dict[str, str]]:
        """
        根据园区和日期获取预约记录。

        Args:
            location: 园区编码
            date: 日期（YYYY-MM-DD）

        Returns:
            预约记录列表
        """
        return self.db.select_by(
            "reservation_records",
            location=location,
            date=date,
        )

    def get_active_count(self, location: str, date: str) -> int:
        """
        获取指定园区在指定日期的有效预约数量。

        Args:
            location: 园区编码
            date: 日期（YYYY-MM-DD）

        Returns:
            有效预约数量（状态为 pending 或 confirmed）
        """
        records = self.get_by_location_date(location, date)
        active = [r for r in records if r.get("status") in ("pending", "confirmed")]
        return len(active)

    def get_remaining_spots(self, location: str, date: str, config: LocationConfig) -> int:
        """
        获取指定园区在指定日期的剩余可用车位。

        Args:
            location: 园区编码
            date: 日期（YYYY-MM-DD）
            config: LocationConfig 实例

        Returns:
            剩余可用车位数
        """
        max_quota = config.get_quota(location, date)
        used = self.get_active_count(location, date)
        remaining = max_quota - used
        return max(0, remaining)
```

```python
        phone: str,
        location: str,
        date: str,
        license_plate: str,
    ) -> Dict[str, Any]:
        """
        处理预约申请。

        进行完整的业务规则校验：
        1. 必填字段检查
        2. 车牌号格式校验
        3. 手机号格式校验
        4. 预约日期范围检查（未来7天内）
        5. 园区状态检查
        6. 配额检查
        7. 重复预约检查

        Args:
            name: 员工姓名
            employee_id: 工号
            phone: 手机号
            location: 园区编码
            date: 预约日期（YYYY-MM-DD）
            license_plate: 车牌号

        Returns:
            包含 success、message、可选 reservation_id 的字典
        """
        try:
            # 1. 必填字段检查
            if not all([name, employee_id, phone, location, date, license_plate]):
                return {"success": False, "message": "所有字段均为必填项"}

            # 2. 车牌号格式校验
            if not validate_license_plate(license_plate):
                return {"success": False, "message": "车牌号格式不合法（示例：鲁G12345 或 鲁G12345E）"}

            # 3. 手机号格式校验
            if not validate_phone(phone):
                return {"success": False, "message": "手机号格式不合法"}

            # 4. 预约日期范围检查
            import datetime as dt_module
            try:
                date_obj = dt_module.datetime.strptime(date, "%Y-%m-%d").date()
            except ValueError:
                return {"success": False, "message": "日期格式不正确，应为 YYYY-MM-DD"}

            today = dt_module.date.today()
            if date_obj < today:
                return {"success": False, "message": "预约日期不能早于今天"}
            max_date = today + dt_module.timedelta(days=self.MAX_FUTURE_DAYS)
            if date_obj > max_date:
                return {"success": False, "message": f"最多可预约未来{self.MAX_FUTURE_DAYS}天内"}

            # 5. 园区状态检查
            location_info = self.location_config.get_location_by_code(location)
            if location_info is None:
                return {"success": False, "message": "园区编码无效"}
            if location_info.get("status") != "active":
                return {"success": False, "message": "该园区暂未开放预约"}

            # 6. 配额检查
            remaining = self.get_remaining_spots(location, date)
            if remaining <= 0:
                return {"success": False, "message": "该园区当日已达预约上限"}

            # 7. 创建预约记录
            success, result = self.reservation_record.create(
                license_plate=license_plate,
                location=location,
                date=date,
                name=name,
                employee_id=employee_id,
                phone=phone,
            )

            if success:
                return {
                    "success": True,
                    "message": "预约成功",
                    "reservation_id": result,
                }
            else:
                return {"success": False, "message": result}

        except Exception as e:
            return {"success": False, "message": f"系统异常，请稍后重试: {str(e)}"}

    def cancel(self, reservation_id: str, license_plate: str) -> Dict[str, Any]:
        """
        处理预约取消。

        Args:
            reservation_id: 预约记录 ID
            license_plate: 车牌号

        Returns:
            包含 success、message 的字典
        """
        try:
            if not reservation_id or not license_plate:
                return {"success": False, "message": "预约ID和车牌号为必填项"}

            # 检查预约记录是否存在
            record = self.reservation_record.get_by_id(reservation_id)
            if record is None:
                return {"success": False, "message": "预约记录不存在"}

            # 检查是否已取消
            if record.get("status") == "canceled":
                return {"success": False, "message": "该预约已被取消，无需重复操作"}

            # 执行取消
            success = self.reservation_record.cancel(reservation_id, license_plate)
            if success:
                return {"success": True, "message": "预约已取消，车位已释放"}
            else:
                return {"success": False, "message": "取消失败，请检查信息是否正确"}

        except Exception as e:
            return {"success": False, "message": f"系统异常，请稍后重试: {str(e)}"}

    def get_reservation_history(
        self, employee_id: str
    ) -> Dict[str, Any]:
        """
        获取指定员工的预约历史记录。

        Args:
            employee_id: 员工工号

        Returns:
            包含 success 和 records 的字典
        """
        try:
            records = self.reservation_record.get_by_employee(employee_id)
            return {"success": True, "records": records}
        except Exception as e:
            return {"success": False, "message": f"查询失败: {str(e)}", "records": []}

    def get_remaining_spots(self, location: str, date: str) -> int:
        """
        获取指定园区在指定日期的剩余可用车位。

        Args:
            location: 园区编码
            date: 日期（YYYY-MM-DD）

        Returns:
            剩余可用车位数
        """
        return self.reservation_record.get_remaining_spots(
            location, date, self.location_config
        )


class PaymentService:
    """
    缴费业务服务类。

    处理提前缴费相关的业务逻辑。
    """

    def __init__(self, db: CsvDatabase) -> None:
        """
        初始化 PaymentService。

        Args:
            db: CsvDatabase 实例
        """
        self.db = db
        self.payment_record = PaymentRecord(db)

    def advance_payment(
        self,
        license_plate: str,
        location: str,
        amount: float,
    ) -> Dict[str, Any]:
        """
        处理提前缴费请求。

        Args:
            license_plate: 车牌号
            location: 园区编码
            amount: 缴费金额

        Returns:
            包含 success、message、可选 payment_id 的字典
        """
        try:
            if not license_plate or not location:
                return {"success": False, "message": "车牌号和园区为必填项"}

            if not validate_license_plate(license_plate):
                return {"success": False, "message": "车牌号格式不合法"}

            if amount <= 0:
                return {"success": False, "message": "缴费金额必须大于0"}

            # 创建缴费记录
            success, result = self.payment_record.create(
                license_plate=license_plate,
                location=location,
                amount=amount,
            )

            if success:
                return {
                    "success": True,
                    "message": "缴费成功，离厂时无需支付",
                    "payment_id": result,
                }
            else:
                return {"success": False, "message": result}

        except Exception as e:
            return {"success": False, "message": f"系统异常，请稍后重试: {str(e)}"}
```
